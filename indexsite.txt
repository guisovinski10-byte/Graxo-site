    // Car medias
    const carMedias = {

    const counterContainer = document.querySelector('.counter-container');

    const carBtns = document.querySelectorAll('[data-cars]');
    const ticketBtns = document.querySelectorAll('[data-ticket]');
    const form = document.getElementById('cta-form');

    // Animate phrases on load
    phrases.forEach((phrase, i) => {
      setTimeout(() => {
        phrase.style.transition = '0.8s var(--ease)';
        phrase.style.opacity = '1';
        phrase.style.transform = 'scale(1) translateY(0)';
      }, i * 1000 + 500);

    // Calculator buttons
    carBtns.forEach(btn => {
      btn.addEventListener('click', () => {
        carBtns.forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        selectedCars = btn.dataset.cars;
        updateLoss();
      });
    });

    ticketBtns.forEach(btn => {
      btn.addEventListener('click', () => {
        ticketBtns.forEach(b => b.classList.remove('active'));
        btn.classList.add('active');
        selectedTicket = parseInt(btn.dataset.ticket);
        updateLoss();
      });
    });

    function updateLoss() {
      if (selectedCars && selectedTicket) {
        currentLoss = carMedias[selectedCars] * selectedTicket * 0.42;
        animateCounter(currentLoss, counterValue);
      }
    }

    // Animate counter
    function animateCounter(target, el = counterValue) {
      const start = parseFloat(el.dataset.current || '0');
      const duration = 1200;
      const startTime = performance.now();

      function step(time) {
        const progress = Math.min((time - startTime) / duration, 1);
        const eased = 1 - Math.pow(1 - progress, 3); // easeOutCubic
        const value = start + (target - start) * eased;
        el.textContent = 'R$ ' + Math.floor(value).toLocaleString('pt-BR');
        el.dataset.current = value;

        if (progress < 1) {
          requestAnimationFrame(step);
        } else {
          el.textContent = 'R$ ' + Math.floor(target).toLocaleString('pt-BR');
          el.dataset.current = target;
        }
      }

      requestAnimationFrame(step);
    }

    // Scroll observers
    const dashboardObserver = new IntersectionObserver((entries) => {

          document.body.classList.add('transformed');

          counterStatus.textContent = 'RECUPERANDO';
